3   import com.beust.jcommander.IParameterValidator;
4   import com.beust.jcommander.ParameterException;
5   
6   import java.util.Locale;
7   import java.util.ResourceBundle;
8   
9   /**
10   * <h1>StatusValidator</h1>
11   * A JCommander validator that check if the given key status is valid.
12   *
13   * @see <a href="http://jcommander.org/#_parameter_validation">JCommander parameter validation</a>
14   */
15  public final class StatusValidator implements IParameterValidator {
16  
17      /**
18       * Validates the given value.
19       *
20       * @param name  The name of the parameter {@link Constants#STATUS_COMPLETE_VALUE} or {@link Constants#STATUS_SHORT_VALUE}.
21       * @param value The value of the parameter that needs to be validated.
22       * @throws ParameterException Thrown if the value of the parameter is invalid.
23       */
24      @Override
25      public void validate(String name, String value) throws ParameterException {
26          if (!value.equals(Constants.PRIMARY) && !value.equals(Constants.ACTIVE) && !value.equals(Constants.INACTIVE))
27              throw new ParameterException(ResourceBundle.getBundle("language", Locale.getDefault()).getString("error.illegal.argument"));
28      }
29  
30  }
