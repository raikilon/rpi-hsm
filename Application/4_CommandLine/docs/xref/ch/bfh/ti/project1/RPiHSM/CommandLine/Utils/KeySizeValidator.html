10  /**
11   * <h1>KeySetSizeValidator</h1>
12   * A JCommander validator that check if the given key size has a valid value.
13   *
14   * @see <a href="http://jcommander.org/#_parameter_validation">JCommander parameter validation</a>
15   */
16  public class KeySizeValidator implements IParameterValidator {
17  
18      /**
19       * Validates the given value.
20       *
21       * @param name  The name of the parameter {@link Constants#SIZE_COMPLETE_VALUE}.
22       * @param value The value of the parameter that needs to be validated.
23       * @throws ParameterException Thrown if the value of the parameter is invalid.
24       */
25      @Override
26      public void validate(String name, String value) throws ParameterException {
27          if (Integer.parseInt(value) < 0 && Integer.parseInt(value) % 2 != 0) { //the key must be even and grater than 0
28              throw new ParameterException(ResourceBundle.getBundle("language", Locale.getDefault()).getString("error.illegal.argument"));
29          }
30      }
31  }
