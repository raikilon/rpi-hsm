3   import ch.bfh.ti.project1.RPiHSM.API.SerialHelper;
4   import ch.bfh.ti.project1.RPiHSM.CommandLine.Utils.Constants;
5   
6   /**
7    * <h1>CommandFactory</h1>
8    * Creates a {@link CommandI} instance.
9    * This factory follows the Command Factory design pattern.
10   */
11  public class CommandFactory {
12  
13      /**
14       * Creates a {@link CommandI} instance by parsing the input value.
15       *
16       * @param command      command to execute (the {@link CommandI} instance)
17       * @param userPath     authenticated user home directory.
18       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
19       * @return a {@link CommandI} instance
20       */
21      public CommandI getCommand(String command, String userPath, SerialHelper serialHelper) {
22          if (command.equals(Constants.SET)) {
23              return new CreateKeySetCommand(userPath, serialHelper);
24          } else if (command.equals(Constants.KEY)) {
25              return new CreateKeyCommand(userPath, serialHelper);
26          } else if (command.equals(Constants.ENCRYPT)) {
27              return new EncryptCommand(userPath, serialHelper);
28          } else if (command.equals(Constants.DECRYPT)) {
29              return new DecryptCommand(userPath, serialHelper);
30          } else if (command.equals(Constants.SIGN)) {
31              return new SignCommand(userPath, serialHelper);
32          } else if (command.equals(Constants.VERIFY)) {
33              return new VerifyCommand(userPath, serialHelper);
34          } else if (command.equals(Constants.CHANGE)) {
35              return new KeyStatusCommand(userPath, serialHelper);
36          } else if (command.equals(Constants.PUB)) {
37              return new PublicKeyCommand(userPath, serialHelper);
38          } else if (command.equals(Constants.DELETE)) {
39              return new DeleteKeySetCommand(userPath, serialHelper);
40          } else if (command.equals(Constants.HELP)) {
41              return new HelpCommand();
42          } else {
43              return new HelpCommand();
44          }
45      }
46  }
