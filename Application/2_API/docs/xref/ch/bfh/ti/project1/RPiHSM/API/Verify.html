3   import javax.naming.OperationNotSupportedException;
4   import java.io.FileNotFoundException;
5   
6   /**
7    * <h1>Verify</h1>
8    * Ask the IoT application using the {@link SerialHelperI} to verify a given signature with a given key set.
9    */
10  public final class Verify {
11  
12      private SerialHelperI serialHelper;
13      private String userPath;
14      private String name;
15      private String file;
16      private String signature;
17  
18      /**
19       * Sets {@link SerialHelperI} instance to communicate with the IoT application and the verify parameters parameters.
20       *
21       * @param serialHelper an instance of {@link SerialHelperI}
22       * @param userPath     home path of the IoT application user
23       * @param name         name of the key set
24       * @param file         absolute path of the original file
25       * @param signature    absolute path of the signature to verify
26       *                     Prerequisite: The given key set must be created with sign purpose.
27       */
28      public Verify(SerialHelperI serialHelper, String userPath, String name, String file, String signature) {
29          this.serialHelper = serialHelper;
30          this.userPath = userPath;
31          this.name = name;
32          this.file = file;
33          this.signature = signature;
34      }
35  
36      /**
37       * Checks if the key set with the {@link CreateKey#name} exists and if yes, it ask to the IoT application to verify the given signature with a given key set.
38       *
39       * @return True if the signature is valid, false if the key set do not exist or the verify process had some errors.
40       * @throws OperationNotSupportedException if an error occurs in the {@link SerialHelperI}
41       * @throws FileNotFoundException          if the given files do not exist
42       */
43      public boolean verify() throws OperationNotSupportedException, FileNotFoundException {
44  
45  
46          KeyExistence ke = new KeyExistence(serialHelper, userPath, name);
47  
48          if (ke.keyexist()) {
49  
50              String[] data = new String[]{Constants.VERIFY, userPath, name};
51              serialHelper.writeLine(String.join(" ", data));
52  
53              serialHelper.sendFile(signature);
54              serialHelper.sendFile(file);
55  
56              if (!serialHelper.status()) {
57                  return false;
58              }
59  
60              return true;
61  
62          } else {
63              return false;
64          }
65      }
66  }
