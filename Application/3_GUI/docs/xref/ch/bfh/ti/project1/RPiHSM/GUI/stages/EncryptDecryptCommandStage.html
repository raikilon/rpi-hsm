1   package ch.bfh.ti.project1.RPiHSM.GUI.stages;
2   
3   import ch.bfh.ti.project1.RPiHSM.API.EncryptDecrypt;
4   import ch.bfh.ti.project1.RPiHSM.API.SerialHelperI;
5   import javafx.scene.control.Button;
6   import javafx.scene.control.Label;
7   import javafx.scene.control.TextField;
8   import javafx.stage.FileChooser;
9   
10  import javax.naming.OperationNotSupportedException;
11  import java.io.File;
12  import java.io.FileNotFoundException;
13  
14  /**
15   * <h1>EncryptDecryptCommandStage</h1>
16   * An implementation of {@link AbstractStage}. This stage allows the user to encrypt or decrypt a file by using the {@link EncryptDecrypt} class.
17   */
18  public class EncryptDecryptCommandStage extends AbstractStage {
19      private Button executeButton, fileChooserButton;
20      private Label keySetLabel, fileLabel, fileNameLabel;
21      private TextField keySetTextField;
22      private FileChooser filechooser;
23      private File file;
24  
25      /**
26       * Generates all the graphical elements for the encrypt or decrypt command.
27       *
28       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
29       * @param userPath     user home directory on the Raspberry Pi
30       * @param encrypt      true to encrypt false to decrypt
31       */
32      public EncryptDecryptCommandStage(SerialHelperI serialHelper, String userPath, boolean encrypt) {
33          super(serialHelper, userPath);
34  
35          //creates the scene objects
36          if (encrypt) sceneTitle = new Label(b.getString("stage.title.encrypt"));
37          else sceneTitle = new Label(b.getString("stage.title.decrypt"));
38          keySetLabel = new Label(b.getString("text.keyset"));
39          keySetTextField = new TextField();
40          fileLabel = new Label(b.getString("text.file"));
41          filechooser = new FileChooser();
42          fileNameLabel = new Label();
43          executeButton = new Button(sceneTitle.getText().contains(b.getString("stage.title.encrypt")) ? b.getString("button.text.encrypt") : b.getString("button.text.decrypt")); //Choices text for the execute button
44          fileChooserButton = new Button(sceneTitle.getText().contains(b.getString("stage.title.encrypt")) ? b.getString("file.chooser.text.file.encrypt") : b.getString("file.chooser.text.file.decrypt")); //Choices text for the filechooser button
45  
46          //On mouse clicked checks, using the stage title,if the file needs to be encrypted or decrypted, then tries to perform the action and displays the success or error messages
47          executeButton.setOnMouseClicked(e1 -> {
48              if (file == null) { //checks if a file has been selected
49                  error(b.getString("file.chooser.text.file.not.chosen"));
50              } else {
51                  EncryptDecrypt ed = new EncryptDecrypt(serialHelper, userPath, keySetTextField.getText(), file.getAbsolutePath());
52  
53                  if (sceneTitle.getText().equals(b.getString("stage.title.encrypt"))) {
54                      try { //begin encrypt
55                          if (ed.encrypt()) {
56                              clearElements();
57                              success(b.getString("command.success.encrypt"));
58                          } else {
59                              error(b.getString("command.error.encrypt"));
60                          }
61                      } catch (OperationNotSupportedException e) {
62                          error(b.getString("error.unsupported.operation"));
63                      } catch (FileNotFoundException e) {
64                          error(b.getString("error.file.not.found"));
65                      }
66                  } else { //begin decrypt
67                      try {
68                          if (ed.decrypt()) {
69                              clearElements();
70                              success(b.getString("command.success.decrypt"));
71                          } else {
72                              error(b.getString("command.error.decrypt"));
73                          }
74                      } catch (OperationNotSupportedException e) {
75                          error(b.getString("error.unsupported.operation"));
76                      } catch (FileNotFoundException e) {
77                          error(b.getString("error.file.not.found"));
78                      }
79                  }
80              }
81          });
82  
83          //On mouse clicked, sets the file to encrypt/decrypt as the file selected with the file chooser
84          fileChooserButton.setOnMouseClicked(e -> {
85              try {
86                  file = filechooser.showOpenDialog(this);
87                  fileNameLabel.setText(file.getAbsolutePath());
88                  success(b.getString("file.chooser.text.file.chosen"));
89              } catch (Exception e2) {
90                  fileNameLabel.setText(b.getString("file.chooser.text.file.not.chosen"));
91              }
92          });
93  
94          //disables the file chooser if the key set does not exist
95          fileChooserButton.disableProperty().bind(messages.textProperty().isEqualTo(b.getString("command.success.keyset.exist")).not());
96          //disables the execute button if no file has been selected using the file chooser
97          executeButton.disableProperty().bind(messages.textProperty().isEqualTo(b.getString("file.chooser.text.file.chosen")).not());
98  
99          //on focus out checks if the key set exists
100         keySetTextField.focusedProperty().addListener((obs, oldVal, newVal) -> {
101             if (!newVal) {
102                 this.checkKeyExistence(keySetTextField.getText());
103             }
104         });
105 
106         //places the objects in the Grid
107         addHeader(true);
108         grid.setGridLinesVisible(false);
109         grid.add(keySetLabel, 0, 1);
110         grid.add(keySetTextField, 1, 1);
111         grid.add(fileLabel, 0, 2);
112         grid.add(fileChooserButton, 1, 2);
113         grid.add(fileNameLabel, 1, 3);
114         grid.add(executeButton, 1, 4);
115     }
116 
117     /**
118      * Resets the elements to their default values
119      */
120     private void clearElements() {
121         keySetTextField.clear();
122         fileNameLabel.setText(null);
123     }
124 }
