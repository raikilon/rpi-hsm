12  import javax.naming.OperationNotSupportedException;
13  
14  /**
15   * <h1>PromoteDemoteCommandStage</h1>
16   * An implementation of {@link AbstractStage}. This stage allows the user to promote or demote a key by using the {@link KeyStatus} class.
17   */
18  public class PromoteDemoteCommandStage extends AbstractStage {
19      private Button executeButton;
20      private Label keySetLabel, versionLabel, actionLabel;
21      private TextField keySetTextField, versionTextField;
22      private ComboBox<String> actionComboBox;
23  
24      /**
25       * Generates all the graphical elements for the promote/demote key command.
26       *
27       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
28       * @param userPath     user home directory on the Raspberry Pi
29       */
30      public PromoteDemoteCommandStage(SerialHelperI serialHelper, String userPath) {
31          super(serialHelper, userPath);
32  
33          //creates the scene objects
34          this.sceneTitle = new Label(b.getString("stage.title.promote.demote"));
35          executeButton = new Button(PROMOTE+" key");
36          keySetLabel = new Label(b.getString("text.keyset"));
37          keySetTextField = new TextField();
38          versionLabel = new Label(b.getString("text.version"));
39          versionTextField = new TextField();
40          actionComboBox = new ComboBox<>();
41          actionLabel = new Label(b.getString("text.action"));
42  
43          actionComboBox.getItems().addAll(PROMOTE, DEMOTE);
44          actionComboBox.getSelectionModel().selectFirst();
45  
46          executeButton.setDisable(true); //disabled at the beginning
47          //On mouse clicked checks if the action to perform is promote or demote, then tries to perform the action and displays the success or error messages
48          executeButton.setOnMouseClicked(e -> {
49              KeyStatus ks = new KeyStatus(serialHelper, userPath, keySetTextField.getText(), Integer.parseInt(versionTextField.getText()));
50              if (actionComboBox.getValue().equals(PROMOTE)) { //Action: promote
51                  try {
52                      if (ks.promote()) {
53                          clearElements();
54                          success(b.getString("command.success.promote"));
55                      } else {
56                          error(b.getString("command.error.promote"));
57                      }
58                  } catch (OperationNotSupportedException e1) {
59                      error(b.getString("error.unsupported.operation"));
60                  }
61              } else { //Action: demote
62                  try {
63                      if (ks.demote()) {
64                          clearElements();
65                          success(b.getString("command.success.demote"));
66                      } else {
67                          error(b.getString("command.error.demote"));
68                      }
69                  } catch (OperationNotSupportedException e1) {
70                      error(b.getString("error.unsupported.operation"));
71                  }
72              }
73          });
74  
75          //disables the execute button if the key set does not exist of if the key version is not a number
76          versionTextField.textProperty().addListener((obs, oldValue, newValue) -> {
77              executeButton.disableProperty().bind(Bindings
78                      .when((new SimpleBooleanProperty(
79                              versionTextField.getText().matches(NUMBER_FIELD_PATTERN)))
80                              .and(keySetTextField.textProperty().length().greaterThan(2)))
81                      .then(false)
82                      .otherwise(true).or(messages.textProperty().isEqualTo(b.getString("command.success.keyset.exist")).not()));
83          });
84  
85          //on value changed, changes the text of execute button (promote or demote key)
86          actionComboBox.setOnAction(e -> executeButton.setText(actionComboBox.getSelectionModel().getSelectedItem() + " key"));
87  
88          //disables version textfield and action combo box if the key set does not exist
89          versionTextField.disableProperty().bind(messages.textProperty().isEqualTo(b.getString("command.success.keyset.exist")).not());
90          actionComboBox.disableProperty().bind(messages.textProperty().isEqualTo(b.getString("command.success.keyset.exist")).not());
91  
92          //on focus out checks if the key set exists
93          keySetTextField.focusedProperty().addListener((obs, oldVal, newVal) -> {
94              if (!newVal) {
95                  this.checkKeyExistence(keySetTextField.getText());
96              }
97          });
98  
99          //places the objects in the Grid
100         addHeader(true);
101         grid.setGridLinesVisible(false);
102         grid.add(keySetLabel, 0, 0);
103         grid.add(keySetTextField, 1, 0);
104         grid.add(versionLabel, 0, 1);
105         grid.add(versionTextField, 1, 1);
106         grid.add(actionLabel, 0, 2);
107         grid.add(actionComboBox, 1, 2);
108         grid.add(executeButton, 1, 4);
109     }
110 
111     /**
112      * Resets the elements to their default values
113      */
114     private void clearElements() {
115         keySetTextField.clear();
116         versionTextField.clear();
117         actionComboBox.getSelectionModel().select(PROMOTE);
118         executeButton.setText(PROMOTE + " key");
119     }
120 }
